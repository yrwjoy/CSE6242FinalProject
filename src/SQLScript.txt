CREATE TABLE REGION (
ID INTEGER PRIMARY KEY ASC,
NAME TEXT,
SIZE_RANK INTEGER
);

CREATE TABLE HOUSE_VALUE_BY_MONTH (
REGION_ID INTEGER,
HOUSETYPE_ID INTEGER,
TIME INTEGER,
VALUE FLOAT,
PRIMARY KEY (REGION_ID, HOUSETYPE_ID, TIME)
);

CREATE TABLE HOUSE_TYPE (
ID INTEGER PRIMARY KEY ASC,
CATEGORY TEXT,
HOUSE_TYPE TEXT
);
